Now, the data will be rolled up by date and product to create a time series.

#Process Issue to bank data
    ts <- sqldf("SELECT Datereceived, Product as product, count(Datereceived) count
                   FROM data
                   GROUP BY Datereceived,product")
## Loading required package: tcltk

The data is in long form. In order to put it into a panel time series, the data needs to be transformed to wide form.

    w <- reshape(ts, 
                 timevar = "product",
                 idvar = c("Datereceived"),
                 direction = "wide")
    
    colnames(w)<-gsub("count.","",colnames(w))
    colnames(w)<-gsub(" ","_",colnames(w))  
  
  w[is.na(w)]<-0
  w$Datereceived <- as.Date(w$Datereceived,"%m/%d/%Y")
  w<-w[order(w$Datereceived),]

  ##Put data into extensible time series format in order to use the Dygraphs library
  graph <- as.xts(w,order.by = w$Datereceived)

In addition, we can take a look at the data in ‘calendar form’.

  calendar_data<-sqldf("SELECT Datereceived as date, count(Datereceived) count
               FROM data
                   GROUP BY date")
  calendar_data$date <- as.Date(calendar_data$date,"%m/%d/%Y")
  calendar_data <- calendar_data[ order(calendar_data$date),]

  Cal <- gvisCalendar(calendar_data, 
                    datevar="date", 
                    numvar="count",
                    options=list(
                      title="Daily complaints to CFPB",
           
                      calendar="{yearLabel: { fontName: 'Helvetica',
                               fontSize: 32, color: 'red', bold: true},
                               cellSize: 10,
                               cellColor: { stroke: 'grey', strokeOpacity: 0.2 },
                               focusedCellColor: {stroke:'grey'}}")
)
plot(Cal)
